解法1 时间复杂度 O(log10(n)) 空间复杂度 O(1)
bool isPalindrome(int x) {
    if(x < 0 ||x % 10 == 0 && x != 0) return false;
    int revertedNumber = 0,number = x;
    while(number) {
        revertedNumber = revertedNumber * 10 + number % 10;
        number /= 10;
    }
    return x == revertedNumber;  
}
