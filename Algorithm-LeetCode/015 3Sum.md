# LeetCode 015
## 3Sum
题目链接：https://leetcode-cn.com/problems/3sum/
解法1 语言:C 时间复杂度 O(nlogn) 空间复杂度 O(n)
```
void mergearray(int *a, int first, int mid, int last, int *temp) {  
    int i = first, j = mid + 1, m = mid, n = last, k = 0;    
    while (i <= m && j <= n) {  
        if (a[i] <= a[j])  
            temp[k++] = a[i++];  
        else  
            temp[k++] = a[j++];  
    }    
    while (i <= m)  
        temp[k++] = a[i++];     
    while (j <= n)  
        temp[k++] = a[j++];  
    for (i = 0; i < k; i++)  
        a[first + i] = temp[i];  
}  
void mergesort(int *a, int first, int last, int *temp) {  
    if (first < last) {  
        int mid = (first + last) / 2;  
        mergesort(a, first, mid, temp);  
        mergesort(a, mid + 1, last, temp); 
        mergearray(a, first, mid, last, temp);  
    }
}
void mergeSort(int *a, int length) {
    int *temp = malloc(sizeof(int) * length);
    mergesort(a, 0, length - 1, temp);
    free(temp);
}
int **threeSum(int *nums, int numsSize, int *returnSize) {
    int i, j, k, m;
    int **ret = malloc(sizeof(int) * (numsSize*(numsSize-1)*(numsSize-2))/6);
    *returnSize = 0;
    if (numsSize == 0) return ret;
    mergeSort(nums,numsSize);
    for (i = 0; i < numsSize; i++) {
        if (i > 0 && nums[i - 1] == nums[i]) {
            continue;
        }
        j = i + 1;
        k = numsSize - 1;
        while (j < k) {
            if (nums[j] + nums[k] + nums[i] == 0) {
                ret[*returnSize] = (int *)malloc(sizeof(int) * 3);
                ret[*returnSize][0] = nums[i];
                ret[*returnSize][1] = nums[j];
                ret[*returnSize][2] = nums[k];
                j++;
                k--;
                (*returnSize)++;
                while (j < k && nums[j] == nums[j - 1]) {
                    j++;
                }
                while (j < k && nums[k] == nums[k + 1]) {
                    k--;
                }
                continue;
            }
            if (nums[j] + nums[k] + nums[i] > 0) {
                k--;
                continue;
            }
            if (nums[j] + nums[k] + nums[i] < 0) {
                j++;
                continue;
            }

        }
    }
    return ret;
}
```
